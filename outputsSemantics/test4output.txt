JUMP main


main:
	START

	// writeln
	PUSHS "Introduza um número inteiro positivo:"
	WRITES
	WRITELN

	// readln
	PUSHL 0
	READI
	READLN
	STOREL 0

	// assign 1 to primo
	PUSHI 1
	STOREL 1

	// assign 2 to i
	PUSHI 2
	STOREL 2

	// If case: {'case': 'primo', 'do': ('writeln', ['num', ' é um número primo']), 'else': ('writeln', ['num', ' não é um número primo'])}
	PUSHL 1
	JZ labelElse1
	// writeln
	PUSHL 0
	WRITEI
	PUSHS " é um número primo"
	WRITES
	WRITELN

	JUMP labelEndIF1
	labelElse1:
	// writeln
	PUSHL 0
	WRITEI
	PUSHS " não é um número primo"
	WRITES
	WRITELN

	labelEndIF1:
	STOP
