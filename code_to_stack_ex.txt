# codigo maquina
// Initialize frame pointer
START                   

// Assign 5 to num1
PUSHI 5                 // Push integer 5
STOREL 0                // Store in fp[0] (num1)

// Assign 7 to num2
PUSHI 7                 //  Push integer 7
STOREL 1                // Store in fp[1] (num2)

// Now perform addition
PUSHL 0                 // Load num1
PUSHL 1                 // Load num2
ADD                     // num1 + num2
WRITEI                  // Output result

STOP                    // End program
